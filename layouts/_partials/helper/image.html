/// Override of themes/hugo-theme-stack/layouts/_partials/helper/image.html /// Reason: 远程图片(R2/CDN) URL 直接透传, 不调用 GetRemote, 避免构建慢和 timeout {{ $result := dict }} {{ if .Image }} {{ $url := urls.Parse .Image }} {{ $permalink := .Image }} {{ $resource := dict "Resource" nil "Permalink" $permalink "Height" nil "Width" nil }} {{ $local := true }} {{ $remote := false }} {{ if not (in (slice "https" "http") $url.Scheme) }} {{/* Local image */}} {{ $resource = .Resources.Get (printf "%s" (.Image | safeURL)) }} {{ else }} {{/* Remote image: 直接透传 URL, 不下载 */}} {{ $remote = true }} {{ $result = dict "Resource" nil "Permalink" $permalink "Height" nil "Width" nil }} {{ end }} {{ if and (not $remote) $resource }} {{ if and .Context.Site.Params.imageProcessing.autoOrient (reflect.IsImageResourceProcessable $resource) }} {{ $filter := images.AutoOrient }} {{ $resource = $resource | images.Filter $filter }} {{ end }} {{ if $local }} {{ $permalink = $resource.RelPermalink }} {{ end }} {{ if reflect.IsImageResourceWithMeta $resource }} {{ $result = dict "Resource" $resource "Permalink" $permalink "Height" $resource.Height "Width" $resource.Width }} {{ else }} {{ $result = dict "Resource" $resource "Permalink" $permalink "Height" nil "Width" nil }} {{ end }} {{ end }} {{ end }} {{ return $result }}
